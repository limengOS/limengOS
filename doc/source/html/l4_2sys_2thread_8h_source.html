751 L4_INLINE l4_msgtag_t
752 l4_thread_ex_regs_u(l4_cap_idx_t thread, l4_addr_t ip, l4_addr_t sp,
753  l4_umword_t flags, l4_utcb_t *utcb) L4_NOTHROW
754 {
755  l4_msg_regs_t *v = l4_utcb_mr_u(utcb);
756  v->mr[0] = L4_THREAD_EX_REGS_OP | flags;
757  v->mr[1] = ip;
758  v->mr[2] = sp;
759  return l4_ipc_call(thread, utcb, l4_msgtag(L4_PROTO_THREAD, 3, 0, 0), L4_IPC_NEVER);
760 }

762 L4_INLINE l4_msgtag_t
763 l4_thread_ex_regs_ret_u(l4_cap_idx_t thread, l4_addr_t *ip, l4_addr_t *sp,
764  l4_umword_t *flags, l4_utcb_t *utcb) L4_NOTHROW
765 {
766  l4_msg_regs_t *v = l4_utcb_mr_u(utcb);
767  l4_msgtag_t ret = l4_thread_ex_regs_u(thread, *ip, *sp, *flags, utcb);
768  if (l4_error_u(ret, utcb))
769  return ret;
770 
771  *flags = v->mr[0];
772  *ip = v->mr[1];
773  *sp = v->mr[2];
774  return ret;
775 }

802 l4_thread_control_bind_u(l4_utcb_t *thread_utcb, l4_cap_idx_t task,
803  l4_utcb_t *utcb) L4_NOTHROW
804 {
805  l4_msg_regs_t *v = l4_utcb_mr_u(utcb);
806  v->mr[L4_THREAD_CONTROL_MR_IDX_FLAGS] |= L4_THREAD_CONTROL_BIND_TASK;
807  v->mr[L4_THREAD_CONTROL_MR_IDX_BIND_UTCB] = (l4_addr_t)thread_utcb;
