162 L4_INLINE long
163 fiasco_ldt_set(l4_cap_idx_t task, void *ldt, unsigned int num_desc,
164  unsigned int entry_number_start, l4_utcb_t *utcb)
165 {
166  if (num_desc > L4_TASK_LDT_X86_MAX_ENTRIES)
167  return -L4_EINVAL;
168  l4_utcb_mr_u(utcb)->mr[0] = L4_TASK_LDT_SET_X86_OP;
169  l4_utcb_mr_u(utcb)->mr[1] = entry_number_start;
170  __builtin_memcpy(&l4_utcb_mr_u(utcb)->mr[2], ldt,
171  num_desc * L4_TASK_LDT_X86_ENTRY_SIZE);
172  return l4_error_u(l4_ipc_call(task, utcb, l4_msgtag(L4_PROTO_TASK, 2 + num_desc * 2, 0, 0), L4_IPC_NEVER), utcb);
173 }

175 L4_INLINE unsigned
176 fiasco_gdt_get_entry_offset(l4_cap_idx_t thread, l4_utcb_t *utcb)
177 {
178  l4_utcb_mr_u(utcb)->mr[0] = L4_THREAD_X86_GDT_OP;
179  if (l4_error_u(l4_ipc_call(thread, utcb, l4_msgtag(L4_PROTO_THREAD, 1, 0, 0), L4_IPC_NEVER), utcb))
180  return -1;
181  return l4_utcb_mr_u(utcb)->mr[0];
182 }

184 L4_INLINE long
185 fiasco_amd64_segment_info(l4_cap_idx_t thread, unsigned *user_ds,
186  unsigned *user_cs, unsigned *user32_cs,
187  l4_utcb_t *utcb)
188 {
189  l4_msg_regs_t *m = l4_utcb_mr_u(utcb);
190  int r;
191 
192  m->mr[0] = L4_THREAD_AMD64_GET_SEGMENT_INFO_OP;
193 
194  r = l4_error_u(l4_ipc_call(thread, utcb, l4_msgtag(L4_PROTO_THREAD, 1, 0, 0),
195  L4_IPC_NEVER), utcb);
196  if (r < 0)
197  return r;
198 
199  *user_ds = m->mr[0];
200  *user_cs = m->mr[1];
201  *user32_cs = m->mr[2];
202 
203  return 0;
204 }
