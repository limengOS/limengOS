236 L4_INLINE l4_msgtag_t
237 l4_irq_mux_chain_u(l4_cap_idx_t irq, l4_cap_idx_t slave,
238  l4_utcb_t *utcb) L4_NOTHROW
239 {
240  l4_msg_regs_t *m = l4_utcb_mr_u(utcb);
241  m->mr[0] = L4_IRQ_MUX_OP_CHAIN;
242  m->mr[1] = l4_map_obj_control(0, 0);
243  m->mr[2] = l4_obj_fpage(slave, 0, L4_CAP_FPAGE_RWS).raw;
244  return l4_ipc_call(irq, utcb, l4_msgtag(L4_PROTO_IRQ_MUX, 1, 1, 0),
245  L4_IPC_NEVER);
246 }
