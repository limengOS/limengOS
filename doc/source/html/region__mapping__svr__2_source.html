220  long op_page_fault(L4::Pager::Rights, l4_umword_t addr, l4_umword_t pc,
221  L4::Ipc::Opt<L4::Ipc::Snd_fpage> &fp)
222  {
223  Dbg(Dbg::Server).printf("page fault: %lx pc=%lx\n", addr, pc);
224 
225  bool need_w = addr & 2;
226  bool need_x = addr & 4;
227 
228  typename DERIVED::Node n = rm()->find(addr);
229 
230  if (!n || !n->second.memory())
231  {
232  Dbg(Dbg::Warn, "rm").printf("unhandled %s page fault at 0x%lx pc=0x%lx\n",
233  need_w ? "write" :
234  need_x ? "instruction" : "read", addr, pc);
235  // generate exception
236  return -L4_ENOMEM;
237  }
238 
239  if (!(n->second.flags() & L4Re::Rm::F::W) && need_w)
240  {
241  Dbg(Dbg::Warn, "rm").printf("write page fault in readonly region at 0x%lx pc=0x%lx\n",
242  addr, pc);
243  // generate exception
244  return -L4_EACCESS;
245  }
246 
247  if (!(n->second.flags() & L4Re::Rm::F::X) && need_x)
248  {
249  Dbg(Dbg::Warn, "rm").printf("instruction page fault in non-exec region at 0x%lx pc=0x%lx\n",
250  addr, pc);
251  // generate exception
252  return -L4_EACCESS;
253  }
254 
255  typename DERIVED::Region_handler::Ops::Map_result map_res;
256  if (int err = n->second.map(addr, n->first, need_w, &map_res))
257  {
258  Dbg(Dbg::Warn, "rm").printf("mapping for page fault failed with error %d at 0x%lx pc=0x%lx\n",
259  err, addr, pc);
260  // generate exception
261  return -L4_ENOMEM;
262  }
263 
264  pager_set_result(&fp, map_res);
265  return L4_EOK;
266  }
