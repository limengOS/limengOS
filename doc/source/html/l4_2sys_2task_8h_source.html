285 L4_INLINE l4_msgtag_t
286 l4_task_map_u(l4_cap_idx_t dst_task, l4_cap_idx_t src_task,
287  l4_fpage_t snd_fpage, l4_umword_t snd_base, l4_utcb_t *u) L4_NOTHROW
288 {
289  l4_msg_regs_t *v = l4_utcb_mr_u(u);
290  v->mr[0] = L4_TASK_MAP_OP;
291  v->mr[3] = l4_map_obj_control(0,0);
292  v->mr[4] = l4_obj_fpage(src_task, 0, L4_CAP_FPAGE_RWS).raw;
293  v->mr[1] = snd_base;
294  v->mr[2] = snd_fpage.raw;
295  return l4_ipc_call(dst_task, u, l4_msgtag(L4_PROTO_TASK, 3, 1, 0), L4_IPC_NEVER);
296 }

309 L4_INLINE l4_msgtag_t
310 l4_task_unmap_batch_u(l4_cap_idx_t task, l4_fpage_t const *fpages,
311  unsigned num_fpages, l4_umword_t map_mask,
312  l4_utcb_t *u) L4_NOTHROW
313 {
314  l4_msg_regs_t *v = l4_utcb_mr_u(u);
315  v->mr[0] = L4_TASK_UNMAP_OP;
316  v->mr[1] = map_mask;
317  __builtin_memcpy(&v->mr[2], fpages, num_fpages * sizeof(l4_fpage_t));
318  return l4_ipc_call(task, u, l4_msgtag(L4_PROTO_TASK, 2 + num_fpages, 0, 0), L4_IPC_NEVER);
