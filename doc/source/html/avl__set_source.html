413 template< typename Item, class Compare, template< typename A > class Alloc, typename KEY_TYPE>
414 Pair<typename Base_avl_set<Item,Compare,Alloc,KEY_TYPE>::Iterator, int>
415 Base_avl_set<Item,Compare,Alloc,KEY_TYPE>::insert(Item const &item)
416 {
417  _Node *n = _alloc.alloc();

419  return cxx::pair(end(), -E_nomem);

422  Pair<_Node *, bool> err = _tree.insert(n);
423  if (!err.second)
