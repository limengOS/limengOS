303 L4_INLINE l4_msgtag_t
304 l4_vcon_send_u(l4_cap_idx_t vcon, char const *buf, unsigned size, l4_utcb_t *utcb) L4_NOTHROW
305 {
306  l4_msg_regs_t *mr = l4_utcb_mr_u(utcb);
307  mr->mr[0] = L4_VCON_WRITE_OP;
308  mr->mr[1] = size;
309  __builtin_memcpy(&mr->mr[2], buf, size);
310  return l4_ipc_send(vcon, utcb,
311  l4_msgtag(L4_PROTO_LOG, 2 + l4_bytes_to_mwords(size),
312  0, L4_MSGTAG_SCHEDULE),
313  L4_IPC_NEVER);
314 }

322 L4_INLINE long
323 l4_vcon_write_u(l4_cap_idx_t vcon, char const *buf, unsigned size, l4_utcb_t *utcb) L4_NOTHROW
324 {
325  l4_msgtag_t t;
326 
327  if (size > L4_VCON_WRITE_SIZE)
328  size = L4_VCON_WRITE_SIZE;
329 
330  t = l4_vcon_send_u(vcon, buf, size, utcb);
331  if (l4_msgtag_has_error(t))
332  return l4_error(t);
333 
334  return (long) size;
335 }

343 L4_INLINE int
344 l4_vcon_read_with_flags_u(l4_cap_idx_t vcon, char *buf, unsigned size,
345  l4_utcb_t *utcb) L4_NOTHROW
346 {
347  int ret;
348  unsigned r;
349  l4_msg_regs_t *mr;
350 
351  mr = l4_utcb_mr_u(utcb);
352  mr->mr[0] = (size << 16) | L4_VCON_READ_OP;
353 
354  ret = l4_error_u(l4_ipc_call(vcon, utcb,
355  l4_msgtag(L4_PROTO_LOG, 1, 0, 0),
356  L4_IPC_NEVER),
357  utcb);
358  if (ret < 0)
359  return ret;
360 
361  r = mr->mr[0] & L4_VCON_READ_SIZE_MASK;
362 
363  if (!(mr->mr[0] & L4_VCON_READ_STAT_DONE)) // !eof
364  ret = size + 1;
365  else if (r < size)
366  ret = r;
367  else
368  ret = size;
369 
370  if (L4_LIKELY(buf != NULL))
371  __builtin_memcpy(buf, &mr->mr[1], r < size ? r : size);
372 
373  return ret | (mr->mr[0] & ~(L4_VCON_READ_STAT_DONE | L4_VCON_READ_SIZE_MASK));
374 }
