642 L4_INLINE l4_fpage_t
643 __l4_fpage_generic(unsigned long address, unsigned int type,
644  unsigned int size, unsigned char rights) L4_NOTHROW
645 {
646  l4_fpage_t t;
647  t.raw = ((rights << L4_FPAGE_RIGHTS_SHIFT) & L4_FPAGE_RIGHTS_MASK)
648  | ((type << L4_FPAGE_TYPE_SHIFT) & L4_FPAGE_TYPE_MASK)
649  | ((size << L4_FPAGE_SIZE_SHIFT) & L4_FPAGE_SIZE_MASK)
650  | ((address ) & L4_FPAGE_ADDR_MASK);
651  return t;
652 }

655 l4_fpage_set_rights(l4_fpage_t src, unsigned char new_rights) L4_NOTHROW
656 {
657  l4_fpage_t f;
658  f.raw = ((L4_FPAGE_TYPE_MASK | L4_FPAGE_SIZE_MASK | L4_FPAGE_ADDR_MASK) & src.raw)
659  | ((new_rights << L4_FPAGE_RIGHTS_SHIFT) & L4_FPAGE_RIGHTS_MASK);
660  return f;

715 L4_INLINE int
716 l4_fpage_contains(l4_fpage_t fpage, l4_addr_t addr, unsigned log2size) L4_NOTHROW
717 {
718  l4_addr_t fa = l4_fpage_memaddr(fpage);
719  return (fa <= addr)
720  && (fa + (1UL << l4_fpage_size(fpage)) >= addr + (1UL << log2size));

723 L4_INLINE unsigned char
724 l4_fpage_max_order(unsigned char order, l4_addr_t addr,
725  l4_addr_t min_addr, l4_addr_t max_addr,
726  l4_addr_t hotspot)
727 {
728  while (order < 30 /* limit to 1GB flexpages */)
729  {
730  l4_addr_t mask;
731  l4_addr_t base = l4_trunc_size(addr, order + 1);
732  if (base < min_addr)
733  return order;
734 
735  if (base + (1UL << (order + 1)) - 1 > max_addr - 1)
736  return order;
737 
738  mask = ~(~0UL << (order + 1));
739  if (hotspot == ~0UL || ((addr ^ hotspot) & mask))
740  break;
741 
742  ++order;
743  }
744 
745  return order;
746 }
