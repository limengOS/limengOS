358 L4_INLINE l4_msgtag_t
359 l4_factory_create_gate_u(l4_cap_idx_t factory,
360  l4_cap_idx_t target_cap,
361  l4_cap_idx_t thread_cap, l4_umword_t label,
362  l4_utcb_t *u) L4_NOTHROW
363 {
364  l4_msgtag_t t;
365  l4_msg_regs_t *v;
366  int items = 0;
367  t = l4_factory_create_start_u(0, target_cap, u);
368  l4_factory_create_add_uint_u(label, &t, u);
369  v = l4_utcb_mr_u(u);
370  if (!(thread_cap & L4_INVALID_CAP_BIT))
371  {
372  items = 1;
373  v->mr[3] = l4_map_obj_control(0,0);
374  v->mr[4] = l4_obj_fpage(thread_cap, 0, L4_CAP_FPAGE_RWS).raw;
375  }
376  t = l4_msgtag(l4_msgtag_label(t), l4_msgtag_words(t), items, l4_msgtag_flags(t));
377  return l4_factory_create_commit_u(factory, t, u);
378 }

455 L4_INLINE int
456 l4_factory_create_add_fpage_u(l4_fpage_t d, l4_msgtag_t *tag,
457  l4_utcb_t *u) L4_NOTHROW
458 {
459  l4_msg_regs_t *v = l4_utcb_mr_u(u);
460  int w = l4_msgtag_words(*tag);
461  if (w + 2 > L4_UTCB_GENERIC_DATA_SIZE)
462  return 0;
463  v->mr[w] = L4_VARG_TYPE_FPAGE | (sizeof(l4_fpage_t) << 16);
464  v->mr[w + 1] = d.raw;
465  w += 2;
466  tag->raw = (tag->raw & ~0x3fUL) | (w & 0x3f);
467  return 1;
468 }

507 L4_INLINE int
508 l4_factory_create_add_lstr_u(char const *s, unsigned len, l4_msgtag_t *tag,
509  l4_utcb_t *u) L4_NOTHROW
510 {
511 
512  l4_msg_regs_t *v = l4_utcb_mr_u(u);
513  unsigned w = l4_msgtag_words(*tag);
514  char *c;
515  unsigned i;
516 
517  if (w + 1 + l4_bytes_to_mwords(len) > L4_UTCB_GENERIC_DATA_SIZE)
518  return 0;
519 
520  v->mr[w] = L4_VARG_TYPE_STRING | (len << 16);
521  c = (char*)&v->mr[w + 1];
522  for (i = 0; i < len; ++i)
523  *c++ = *s++;
524 
525  w = w + 1 + l4_bytes_to_mwords(len);
526 
527  tag->raw = (tag->raw & ~0x3fUL) | (w & 0x3f);
528  return 1;
529 }
