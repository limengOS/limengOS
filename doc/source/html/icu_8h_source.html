392 L4_INLINE l4_msgtag_t

396  l4_msg_regs_t *m = l4_utcb_mr_u(utcb);

430 L4_INLINE l4_msgtag_t
431 l4_icu_msi_info_u(l4_cap_idx_t icu, unsigned irqnum, l4_uint64_t source,
432  l4_icu_msi_info_t *msi_info, l4_utcb_t *utcb) L4_NOTHROW
433 {
434  l4_msgtag_t res;
435  l4_msg_regs_t *m = l4_utcb_mr_u(utcb);
436  m->mr[0] = L4_ICU_OP_MSI_INFO;
437  m->mr[1] = irqnum;
438  m->mr64[l4_utcb_mr64_idx(2)] = source;
439  res = l4_ipc_call(icu, utcb, l4_msgtag(L4_PROTO_IRQ,
440  2 + 1 * sizeof(l4_uint64_t)
441  / sizeof(l4_umword_t),
442  0, 0), L4_IPC_NEVER);
443  if (L4_UNLIKELY(l4_msgtag_has_error(res)))
444  return res;
445 
446  if (L4_UNLIKELY(l4_msgtag_words(res) * sizeof(l4_umword_t) < sizeof(*msi_info)))
447  return res;
448 
449  __builtin_memcpy(msi_info, &m->mr[0], sizeof(*msi_info));
450  return res;
451 }

465 l4_icu_control_u(l4_cap_idx_t icu, unsigned irqnum, unsigned op,
466  l4_umword_t *label, l4_timeout_t to,
467  l4_utcb_t *utcb) L4_NOTHROW
468 {
469  l4_msg_regs_t *m = l4_utcb_mr_u(utcb);
