319 template< typename L >
320 inline l4_msgtag_t
321 Server<L>::reply_n_wait(l4_msgtag_t reply, l4_umword_t *p, l4_utcb_t *utcb)
322 {
323  if (reply.label() != -L4_ENOREPLY)
324  {
325  Ipc_svr::Reply_mode m = this->before_reply(reply, utcb);
326  if (m == Ipc_svr::Reply_compound)
327  {
328  this->setup_wait(utcb, m);
329  return l4_ipc_reply_and_wait(utcb, reply, p, this->timeout());
330  }
331  else
332  {
333  l4_msgtag_t res = l4_ipc_send(L4_INVALID_CAP | L4_SYSF_REPLY, utcb, reply, this->timeout());
334  if (res.has_error())
335  return res;
336  }
337  }
338  this->setup_wait(utcb, Ipc_svr::Reply_separate);
339  return l4_ipc_wait(utcb, p, this->timeout());
340 }

342 template< typename L >
343 template< typename DISPATCH >
344 inline L4_NORETURN void
345 Server<L>::internal_loop(DISPATCH dispatch, l4_utcb_t *utcb)
346 {
347  l4_msgtag_t res;
348  l4_umword_t p;
349  l4_msgtag_t r = l4_msgtag(-L4_ENOREPLY, 0, 0, 0);
350 
351  while (true)
352  {
353  res = reply_n_wait(r, &p, utcb);
354  if (res.has_error())
355  {
356  this->error(res, utcb);
357  r = l4_msgtag(-L4_ENOREPLY, 0, 0, 0);
358  continue;
359  }
360 
361  r = dispatch(res, p, utcb);
362  }
363 }
