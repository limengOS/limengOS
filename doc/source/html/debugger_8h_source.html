231 __strcpy_maxlen(char *dst, char const *src, unsigned maxlen)
232 {
233  unsigned i;
234  if (!maxlen)
235  return 0;
236 
237  for (i = 0; i < maxlen - 1 && src[i]; ++i)

241  return i + 1;
