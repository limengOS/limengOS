264 L4_INLINE l4_msgtag_t
265 l4_scheduler_info_u(l4_cap_idx_t scheduler, l4_umword_t *cpu_max,
266  l4_sched_cpu_set_t *cpus, l4_umword_t *sched_classes,
267  l4_utcb_t *utcb) L4_NOTHROW
268 {
269  l4_msg_regs_t *m = l4_utcb_mr_u(utcb);
270  l4_msgtag_t res;
271 
272  m->mr[0] = L4_SCHEDULER_INFO_OP;
273  m->mr[1] = cpus->gran_offset;
274 
275  res = l4_ipc_call(scheduler, utcb, l4_msgtag(L4_PROTO_SCHEDULER, 2, 0, 0), L4_IPC_NEVER);
276 
277  if (l4_msgtag_has_error(res))
278  return res;
279 
280  cpus->map = m->mr[0];
281 
282  if (cpu_max)
283  *cpu_max = m->mr[1];
284 
285  if (sched_classes)
286  *sched_classes = m->mr[2];
287 
288  return res;
289 }

291 L4_INLINE l4_msgtag_t
292 l4_scheduler_run_thread_u(l4_cap_idx_t scheduler, l4_cap_idx_t thread,
293  l4_sched_param_t const *sp, l4_utcb_t *utcb) L4_NOTHROW
294 {
295  l4_msg_regs_t *m = l4_utcb_mr_u(utcb);
296  m->mr[0] = L4_SCHEDULER_RUN_THREAD_OP;
297  m->mr[1] = sp->affinity.gran_offset;
298  m->mr[2] = sp->affinity.map;
299  m->mr[3] = sp->prio;
300  m->mr[4] = sp->quantum;
301  m->mr[5] = l4_map_obj_control(0, 0);
302  m->mr[6] = l4_obj_fpage(thread, 0, L4_CAP_FPAGE_RWS).raw;
303 
304  return l4_ipc_call(scheduler, utcb, l4_msgtag(L4_PROTO_SCHEDULER, 5, 1, 0), L4_IPC_NEVER);
305 }

308 l4_scheduler_idle_time_u(l4_cap_idx_t scheduler, l4_sched_cpu_set_t const *cpus,
309  l4_kernel_clock_t *us, l4_utcb_t *utcb) L4_NOTHROW

316  v->mr[2] = cpus->map;
317 
318  res = l4_ipc_call(scheduler, utcb,
319  l4_msgtag(L4_PROTO_SCHEDULER, 3, 0, 0), L4_IPC_NEVER);
320 
321  if (l4_msgtag_has_error(res))
322  return res;
323 
324  *us = v->mr64[l4_utcb_mr64_idx(0)];
325 
326  return res;

330 L4_INLINE int
331 l4_scheduler_is_online_u(l4_cap_idx_t scheduler, l4_umword_t cpu,
332  l4_utcb_t *utcb) L4_NOTHROW
333 {
334  l4_sched_cpu_set_t s;
335  l4_msgtag_t r;

339  return 0;
340 
341  return s.map & 1;
